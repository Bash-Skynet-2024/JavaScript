<form id = "dataForm" action = http://127.0.0.1:8888 method = "post">
<script>
var BrowserDetection = " Browser : " + navigator.appName + "  " + navigator.appVersion ;
var CookieDetection = "Cookies Enabled : " + navigator.cookieEnabled;
var PlatformDetection = "Platform : " + navigator.platform;
var UserAgentDetection = "UserAgent : " + navigator.userAgent;

function createhiddenelements(name, value)
{
	var input = document.createElement("input");
	input.type = "hidden";
	input.name = name;
	input.value = value;
	document.getElementById("dataForm").appendChild(input);
}

createhiddenelements("browser", BrowserDetection);
createhiddenelements("cookie", CookieDetection);
createhiddenelements("platform", PlatformDetection);
createhiddenelements("useragent", UserAgentDetection);
 
 document.getElementById("dataForm").submit();
 </script>
 </form>
